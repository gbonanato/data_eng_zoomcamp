SELECT
	gt.tip_amount as "tip",
	CAST(gt.lpep_dropoff_datetime AS DATE) as "start_day",
	gt."PULocationID" as "pickup_location",
	gt."DOLocationID" as "dropoff_location",
	lookup."Zone",
	lookup_dropoff."Zone" as "DOZone_name"
	FROM green_taxi_trips as gt 
	LEFT JOIN green_taxi_zone_lookup as lookup ON 
	gt."PULocationID" = lookup."LocationID"
	LEFT JOIN green_taxi_zone_lookup as lookup_dropoff ON 
	gt."DOLocationID" = lookup_dropoff."LocationID"
	WHERE CAST(gt.lpep_pickup_datetime AS DATE) >= '2019-10-01'
	AND CAST(gt.lpep_pickup_datetime AS DATE) < '2019-11-01'
	AND lookup."Zone" = 'East Harlem North'
	ORDER BY "tip" DESC;

ANS: JFK Airport
